    <script>
        // --- STATE MANAGEMENT ---
        let logs = JSON.parse(localStorage.getItem('hinoIndosentosaLogsFinal')) || [];
        let currentBranch = "Rancaekek"; // Default Cabang
        
        // Script URL (Default kosong, akan diisi user lewat menu Settings)
        let GOOGLE_SCRIPT_URL = localStorage.getItem('gScriptUrl') || "";

        // --- INIT ---
        document.addEventListener('DOMContentLoaded', () => {
            setDefaults();
            updateClock();
            setInterval(updateClock, 60000);
            checkConnectionStatus();
            document.getElementById('filterDateLog').valueAsDate = new Date();
            
            // Load saved URL to input if exists
            if(GOOGLE_SCRIPT_URL) {
                document.getElementById('scriptUrlInput').value = GOOGLE_SCRIPT_URL;
            }
        });

        function setDefaults() {
            const now = new Date();
            document.getElementById('tanggal').valueAsDate = now;
            document.getElementById('waktu').value = now.toTimeString().substring(0,5);
        }

        function updateClock() {
            const now = new Date();
            const options = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric', hour: '2-digit', minute: '2-digit' };
            document.getElementById('currentDateTime').textContent = now.toLocaleDateString('id-ID', options);
        }

        function checkConnectionStatus() {
            const statusEl = document.getElementById('connectionStatus');
            if (GOOGLE_SCRIPT_URL && GOOGLE_SCRIPT_URL.length > 10) {
                statusEl.textContent = "Terhubung ke Database Online";
                statusEl.className = "connection-status status-connected";
            } else {
                statusEl.textContent = "Tidak Terkoneksi (Hanya Lokal)";
                statusEl.className = "connection-status status-disconnected";
            }
        }

        // --- URL CONFIGURATION ---
        function saveScriptUrl() {
            const url = document.getElementById('scriptUrlInput').value.trim();
            if (url) {
                localStorage.setItem('gScriptUrl', url);
                GOOGLE_SCRIPT_URL = url;
                alert('URL Database berhasil disimpan!');
                checkConnectionStatus();
            } else {
                alert('URL tidak boleh kosong.');
            }
        }

        // --- BRANCH HANDLING ---
        function handleBranchChange() {
            currentBranch = document.getElementById('globalBranch').value;
            if(!document.getElementById('dashboard').classList.contains('hidden')) renderDashboard();
            if(!document.getElementById('data').classList.contains('hidden')) renderLogTable();
        }

        // --- AUTO FILL LOGIC ---
        function autoFillDetails() {
            const plat = document.getElementById('plat').value.toUpperCase();
            if(plat.length < 4) return;
            const lastLog = logs.find(l => l.plat === plat);
            if(lastLog) {
                document.getElementById('noRangka').value = lastLog.noRangka || '';
                document.getElementById('jenisKendaraan').value = lastLog.jenisKendaraan || 'Mobil';
            }
        }

        // --- CORE SAVE LOGIC (ONLINE INTEGRATION) ---
        function handleSubmit(e) {
            e.preventDefault();

            const newLog = {
                id: Date.now(),
                cabang: currentBranch,
                plat: document.getElementById('plat').value.toUpperCase(),
                jenisKendaraan: document.getElementById('jenisKendaraan').value,
                jenisKunjungan: document.getElementById('jenisKunjungan').value,
                keperluan: document.getElementById('keperluan').value,
                tujuan: document.getElementById('tujuan').value,
                tanggal: document.getElementById('tanggal').value,
                waktu: document.getElementById('waktu').value,
                kondisi: document.getElementById('kondisi').value,
                km: document.getElementById('km').value,
                bbm: document.getElementById('bbm').value,
                noRangka: document.getElementById('noRangka').value.toUpperCase(),
                petugas: document.getElementById('petugas').value,
                catatan: document.getElementById('catatan').value,
                timestamp: new Date().getTime()
            };

            // 1. Simpan ke Array Lokal
            logs.unshift(newLog);
            
            // 2. Simpan ke LocalStorage
            localStorage.setItem('hinoIndosentosaLogsFinal', JSON.stringify(logs));
            
            // 3. Kirim ke Google Sheets (Jika URL ada)
            if (GOOGLE_SCRIPT_URL) {
                sendToGoogleSheets(newLog);
            } else {
                alert('Data tersimpan di LOKAL (Browser). \n\nUntuk menyimpan ke Database Online, silakan masukkan URL Google Script di menu "Pengaturan Data".');
            }

            // 4. Refresh UI
            renderDashboard();
            renderLogTable();
            document.getElementById('logForm').reset();
            setDefaults();
        }

        function sendToGoogleSheets(data) {
            const btn = document.querySelector('#logForm button[type="submit"]');
            const originalText = btn.innerHTML;
            btn.innerHTML = '<i class="ph ph-spinner" style="animation: spin 1s linear infinite;"></i> Mengirim...';
            btn.disabled = true;

            fetch(GOOGLE_SCRIPT_URL, {
                method: 'POST',
                mode: 'no-cors', 
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(data)
            })
            .then(() => {
                alert('Data tersimpan di LOKAL dan berhasil dikirim ke DATABASE ONLINE (Google Sheets)!');
            })
            .catch(error => {
                console.error('Error:', error);
                alert('Data tersimpan di LOKAL. Gagal kirim ke Online (Cek koneksi internet atau URL Script).');
            })
            .finally(() => {
                btn.innerHTML = originalText;
                btn.disabled = false;
            });
        }

        // --- DATA MANAGEMENT (BACKUP/RESTORE) ---
        function downloadData() {
            const dataStr = JSON.stringify(logs, null, 2);
            const blob = new Blob([dataStr], { type: "application/json" });
            const url = URL.createObjectURL(blob);
            const filename = `Backup_Hino_${new Date().toISOString().split('T')[0]}.json`;
            const a = document.createElement('a');
            a.href = url;
            a.download = filename;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
        }

        function uploadData(inputElement) {
            const file = inputElement.files[0];
            if (!file) return;
            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    const importedData = JSON.parse(e.target.result);
                    if (Array.isArray(importedData)) {
                        if(confirm('Data lokal akan ditimpa. Lanjutkan?')) {
                            logs = importedData;
                            localStorage.setItem('hinoIndosentosaLogsFinal', JSON.stringify(logs));
                            renderDashboard();
                            renderLogTable();
                            alert('Data berhasil dipulihkan (Restore)!');
                        }
                    } else { alert('Format file salah.'); }
                } catch (err) { alert('File corrupt atau bukan JSON.'); }
                inputElement.value = '';
            };
            reader.readAsText(file);
        }

        function clearAllData() {
            if(confirm('Hapus seluruh data?')) {
                if(confirm('Yakin? Data tidak bisa kembali.')) {
                    logs = [];
                    localStorage.removeItem('hinoIndosentosaLogsFinal');
                    renderDashboard();
                    renderLogTable();
                    alert('Data dihapus.');
                }
            }
        }

        // --- RENDER FUNCTIONS ---
        function renderDashboard() {
            const todayStr = new Date().toISOString().split('T')[0];
            const todayLogs = logs.filter(l => l.cabang === currentBranch && l.tanggal === todayStr);
            const countMasuk = todayLogs.filter(l => l.jenisKunjungan === 'Kendaraan Masuk').length;
            const countKeluar = todayLogs.filter(l => l.jenisKunjungan === 'Kendaraan Keluar').length;
            document.getElementById('statMasuk').textContent = countMasuk;
            document.getElementById('statKeluar').textContent = countKeluar;

            const vehicleStatus = {};
            const branchLogs = logs.filter(l => l.cabang === currentBranch);
            branchLogs.forEach(log => {
                if(!vehicleStatus[log.plat] || log.timestamp > vehicleStatus[log.plat].timestamp) {
                    vehicleStatus[log.plat] = log;
                }
            });

            const menginapList = [];
            for (const plat in vehicleStatus) {
                const lastLog = vehicleStatus[plat];
                if (lastLog.jenisKunjungan === 'Kendaraan Masuk' && lastLog.tanggal < todayStr) {
                    menginapList.push(lastLog);
                }
            }
            document.getElementById('statMenginap').textContent = menginapList.length;

            const tbody = document.querySelector('#menginapTable tbody');
            tbody.innerHTML = '';
            if(menginapList.length === 0) {
                tbody.innerHTML = '<tr><td colspan="5" style="text-align:center; color:#888;">Tidak ada kendaraan menginap.</td></tr>';
            } else {
                menginapList.forEach(item => {
                    tbody.innerHTML += `
                        <tr>
                            <td style="font-weight:bold; color:var(--hino-blue)">${item.plat}</td>
                            <td>${item.jenisKendaraan}</td>
                            <td>${item.keperluan}</td>
                            <td>${item.tanggal} ${item.waktu}</td>
                            <td>${item.petugas}</td>
                        </tr>`;
                });
            }
        }

        function renderLogTable() {
            const filterDate = document.getElementById('filterDateLog').value;
            const tbody = document.querySelector('#logTable tbody');
            tbody.innerHTML = '';
            let displayLogs = logs.filter(l => l.cabang === currentBranch);
            if(filterDate) displayLogs = displayLogs.filter(l => l.tanggal === filterDate);

            if(displayLogs.length === 0) {
                tbody.innerHTML = '<tr><td colspan="14" style="text-align:center; padding:2rem;">Tidak ada data.</td></tr>';
                return;
            }

            displayLogs.forEach(log => {
                const tr = document.createElement('tr');
                const isIn = log.jenisKunjungan === 'Kendaraan Masuk';
                const badgeClass = isIn ? 'badge-in' : 'badge-out';
                tbody.innerHTML += `
                    <tr>
                        <td><div style="font-weight:bold;">${log.waktu}</div><div style="font-size:0.7rem; color:#888;">${log.tanggal}</div></td>
                        <td style="font-weight:bold; color:var(--hino-blue);">${log.plat}</td>
                        <td>${log.jenisKendaraan}</td>
                        <td><span class="status-badge ${badgeClass}">${log.jenisKunjungan}</span></td>
                        <td>${log.keperluan}</td>
                        <td>${log.tujuan || '-'}</td>
                        <td>${log.kondisi}</td>
                        <td>${log.km}</td>
                        <td>${log.bbm}</td>
                        <td style="font-size:0.75rem; font-family:monospace;">${log.noRangka || '-'}</td>
                        <td>${log.petugas}</td>
                        <td style="max-width:150px; white-space:nowrap; overflow:hidden; text-overflow:ellipsis;" title="${log.catatan}">${log.catatan || '-'}</td>
                        <td class="no-print">
                            <button onclick="deleteLog(${log.id})" style="border:none; background:none; color:red; cursor:pointer; font-size:1.2rem;"><i class="ph ph-trash"></i></button>
                        </td>
                    </tr>`;
            });
        }

        function deleteLog(id) {
            if(confirm('Hapus data ini?')) {
                logs = logs.filter(l => l.id !== id);
                localStorage.setItem('hinoIndosentosaLogsFinal', JSON.stringify(logs));
                renderDashboard();
                renderLogTable();
            }
        }

        function switchTab(tabId, navEl) {
            document.querySelectorAll('.section').forEach(s => s.classList.add('hidden'));
            document.getElementById(tabId).classList.remove('hidden');
            document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
            navEl.classList.add('active');

            const titles = { 'input': 'Input Pencatatan', 'dashboard': 'Dashboard Operasional', 'data': 'Data Log', 'settings': 'Pengaturan Admin' };
            document.getElementById('pageTitle').textContent = titles[tabId];

            if(tabId === 'dashboard') renderDashboard();
            if(tabId === 'data') renderLogTable();
        }
